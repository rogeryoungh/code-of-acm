Roger Young 的板子

扩展欧几里得・EXGCD

求出方程

$$ ax + by = \gcd(a, b) $$

的一个解 $(x, y)$，并且求出 $\gcd(a, b)$。

// @description 扩展欧几里得

tuple<ll, ll, ll> exgcd(ll a, ll b) {
	if (b == 0) {
		return {1, 0, a};
	} else {
		auto [y, x, g] = exgcd(b, a % b);
		return {x, y - a / b * x, g};
	}
}

ll inv_gcd(ll a, ll p) {
	return (get<0>(exgcd(a, p)) % p + p) % p;
}
