## 题目大意

在三维空间中，HKE 只能往上走，求攀爬总长。

## 分析

对所有座标的 $z$ 分量排序即可。

```cpp
struct POINT {
    double x, y, z;
    bool operator<(POINT p) {
        return z > p.z;
    }
} nn[50010];

double dis(POINT &p1, POINT &p2) {
    double t1 = p1.x - p2.x, t2 = p1.y - p2.y, t3 = p1.z - p2.z;
    return sqrt(t1 * t1 + t2 * t2 + t3 * t3);
}

int main() {
    int n = rr();
    for (ll i = 1; i <= n; i++)
        cin >> nn[i].x >> nn[i].y >> nn[i].z;
    sort(nn + 1, nn + n + 1);
    double len = 0;
    for (ll i = 2; i <= n; i++)
        len += dis(nn[i], nn[i-1]);
    printf("%.3lf\n", len);
    return 0;
}
```
